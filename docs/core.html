The Crossfader
             		                            
crossfade = (element)->
  x = parseInt(element.value) / parseInt(element.max)
  gain1 = Math.cos(x * 0.5*Math.PI)
  gain2 = Math.cos((1.0 - x) * 0.5*Math.PI)
  deckA.gainNode.gain.value = gain1;
  deckB.gainNode.gain.value = gain2;
